#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n, m;
    cin >> n >> m;

    vector <vector <int>> a(n, vector <int> (m, 1));
    vector <vector <int>> b(n, vector <int> (m));

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> b[i][j];
        }
    }

    for (int i = 1; i < n; i++) {
        for (int j = 1; j < m; j++) {
            if (!b[i][j]) {
                if (b[i][j-1] && b[i-1][j])
                    continue;
                else {
                    if ((!b[i][j-1] && !b[i-1][j]))
                        a[i][j] = a[i][j-1] + a[i-1][j];
                    else {
                        if (!b[i][j-1]) a[i][j] = a[i][j-1];
                        if (!b[i-1][j]) a[i][j] = a[i-1][j];
                    }
                }
            }
        }
    }

    cout << a[n-1][m-1];

    return 0;
}
