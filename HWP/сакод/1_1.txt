#include <iostream>

using namespace std;

int main() {
    long long n;
    cin >> n;
    
    for (long long i = 1; i*i <= n; i++) {
        if (n%i == 0) {
            if (i*i != n)
                cout << i << " " << n/i << endl;
            else
                cout << i;
        }
        
    }

    return 0;
}
