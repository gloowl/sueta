#include <iostream>
#include <vector>

using namespace std;

void dfs(int v, vector<vector<int>>& adj_list, vector<bool>& used) {
    used[v] = true;
    for (int u : adj_list[v]) {
        if (!used[u]) {
            dfs(u, adj_list, used);
        }
    }
}

int countConnectedComponents(vector<vector<int>>& g) {
    int n = g.size();
    vector<vector<int>> adj_list(n);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (g[i][j] == 1) {
                adj_list[i].push_back(j);
                adj_list[j].push_back(i);
            }
        }
    }

    vector<bool> used(n, false);
    int count = 0;
    for (int i = 0; i < n; i++) {
        if (!used[i]) {
            dfs(i, adj_list, used);
            count++;
        }
    }

    return count;
}

int main() {
    int n;
    cin >> n;

    vector<vector<int>> g(n, vector<int>(n));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> g[i][j];
        }
    }

    int num_connected_components = countConnectedComponents(g);
    cout << "Number of connected components: " << num_connected_components << endl;

    return 0;
}
