#include <iostream> 

using namespace std; 

long long binPow(long long a, long long n) { 
    long long res = 1; 
    
    while (n) { 
        if (n % 2 == 1)
            res *= a; 

        a *= a; 

        a %= (long long)(1e9 + 7);

        n /= 2; 

        res %= (long long)(1e9 + 7); 
    } 
    return res; 
} 

int main() 
{ 
    long long  a, n; 
    cin >> a >> n; 

    cout << binPow(a, n) << endl; 

    return 0; 
}
