#include <iostream>
#include <map>
#include <string>
#include <sstream>

using namespace std;

int main() {
    int n;
    cin >> n;

    map <string, string> m;
    map <string, int> count;

    string cur, dd, str;

    for (int i = 0; i < n; i++) {
        cin >> cur >> dd;

        getline(cin, str);
        stringstream inp (str);
        while(inp >> str) {
            m[str] = cur;
        }

        m[cur] = cur;
        count[cur] = 0;
    }

    while (cin >> str) {
        if (m.find(str) != m.end()) {
            count[m[str]]++;
        }
    }

    for (auto i : count) {
        cout << i.first << " " << i.second << endl;
    }

    return 0;
}
