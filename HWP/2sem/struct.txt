#include <iostream>
#include <string>
#include <vector>
#include <iomanip>

using namespace std;

int main() {
    setlocale(LC_ALL, "rus");
    
    struct Toy {
        string name;
        int price;
        int minAge;
        int maxAge;
    };

    vector<Toy> toys = {
        {"Мяч прост", 100, 3, 10},
        {"Кубики", 200, 2, 6},
        {"Кукла", 300, 4, 8},
        {"Мяч прем", 150, 5, 12},
        {"Машинка", 250, 3, 7},
        {"Конструкт", 350, 6, 10},
        {"Каталка", 300, 1, 3},
        {"Плюш мишка", 400, 0, 5},
        {"Динозавр", 200, 4, 9},
        {"Мяч бюдж", 50, 4, 99}
    };

    vector<string> result;

    // ищем подходящие игрушки (мяч + еще одна)
    for (int i = 0; i < toys.size(); i++) {
        if (toys[i].name.find("Мяч") != string::npos) {
            for (int j = 0; j < toys.size(); j++) {
                if (i != j && toys[j].price + toys[i].price <= 500 && toys[j].minAge <= 3 && toys[j].maxAge >= 3 && toys[j].name.find("Мяч") == string::npos) {
                    result.push_back(toys[i].name);
                    result.push_back(toys[j].name);
                }
            }
        }
    }

    // вывод результата
    if (result.empty()) {
        cout << "Подходящих игрушек не найдено\n";
    } else {
        cout << "Найдены подходящие игрушки:\n";
        for (int i=0; i < result.size(); i+=2) {
            cout << "\t- " << result[i] << " | " << result[i+1] << endl;
        }
    }

    cout << "Список всех игрушек:\n";
    cout << setw(12) << "Название" << " Стоимость" <<  " Мин. возраст" << " Макс. возраст\n";
    for (Toy t : toys) {
        cout << setw(12) << t.name << setw(10) << t.price << setw(8) << t.minAge << setw(9) << t.maxAge << endl;
    }

    return 0;
}
