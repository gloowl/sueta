#include <iostream>

using namespace std;

bool checkIsPrime(int num) {

    if (num == 1) return 0;


    for (int i = 2; i*i <= num; i++) {
        if (num % i == 0) return 0;
    }

    return 1;
}

int main() {

    for (;;) {
        cout << "Enter N: \t";

        int N;
        cin >> N;

        
        cout << '\n' << endl;

        for (int i = N+2; i <= 2*N; i++) {
            if (checkIsPrime(i) == 1 && checkIsPrime(i-2) == 1) {
                cout << i-2 << " and " << i << endl;
            }
        }
    }

    return 0;
}
