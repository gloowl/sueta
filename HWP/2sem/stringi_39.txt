/*
Задание 2. Дана строка, содержащая последовательность слов, разделенных пробелами. 
Слова состоят из латинских строчных букв. Произвести со строкой требуемые изменения. 
Результатом должна быть строка, в которой слова разделены одним пробелом.

39.  Добавить после каждого слова его длину, взятую в квадратные скобки. 
*/

#include <iostream>
#include <string>

using namespace std;

int main() {
    setlocale(LC_ALL, "rus");

    // ввод строки
    string str;
    cout << "Введите строку: ";
    getline(cin, str);

    // добавление длины слова в квадратных скобках
    int start = 0;
    for (int i = 0; i < str.length(); i++) {
        // если текущий символ пробел, добавляем длину слова до этого символа в квадратных скобках
        if (str[i] == ' ') {
            int length = i - start;
            str.insert(start + length, "[" + to_string(length) + "]");

            // перепрыгиваем через пробелы
            i += 3 + to_string(length).length();
            start = i;
        }
    }

    // обработка последнего слова в строке, который не содержит пробелов
    int length = str.length() - start;
    str.insert(start + length, "[" + to_string(length) + "]");

    // вывод измененной строки
    cout << "Измененная строка: " << str << endl;

    return 0;
}
