#include <iostream>
#include <math.h>
using namespace std;
int main() {
    do {
        double x;
        cout << "Input X: " << endl;
        cin >> x;
        double y = x;
        int fact(1), f1(1);
        for (int f2 = 1; f2<=5; f2++) {
            fact *= f2;
            if ((pow(3, f2) - y) != 0) { // Вычисление возможно
                y = fact * f1 / (pow(3, f2) - y); 
            }
            else {
                cout << "NO" << endl; // Вычисление не возможно
                return 0;             // Программа завершается
            }
            f1 += 2;
            cout << "y = " << y << endl;
        }
        cout << "\nOutput: y = " << y << '\n' << endl;
    } while (true);
    return 0;
}
