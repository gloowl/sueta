#include <iostream>
#include <stdlib.h>
#include <time.h>
#include <math.h>
using namespace std;
int main(){
    do{
        int N, mn(0), help;
        cout << "Enter N: " << endl;
        cin >> N;
        int X[N];
        srand(time(NULL));
        for(int i = 0; i < N; i++){
            X[i] = rand()%10000-5000;
            cout << X[i] << ' ';
        }
        /*for(int i = 0; i < N; i++){
            cout << "Enter X[" << i << "]:\t";
            cin >> X[i];
        }
        for(int i = 0; i < N; i++){
            cout << X[i] << ' ';
        }*/
        cout << endl;
        for(int a=0; a<N; a++){
            for(int num = abs(X[a]); num; num/=10){
                if(num%10 == 5){
                    mn = X[a];
                    help = a;
                    break;
                }
            }
            if(help == a) break;
        }
        for(int i=help+1; i<N; i++){
            for(int num = abs(X[i]); num; num/=10){
                if(num%10 == 5){
                    if(mn > X[i]){
                        mn = X[i];
                        
                    }
                    break;
                }
            }
        }
        cout << "Output: \t" << mn << endl;
        cout << endl;
    }while(true);
    return 0;
}
